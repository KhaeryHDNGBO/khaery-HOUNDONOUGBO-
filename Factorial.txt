#include <stdio.h>
#include <stdlib.h>
int factorielle (int n)
{
   int i;    // compteur de boucle //
   if (n == 0)
       {
       return 1;
       }
  else
      {
       for ( i = 1; i <= n-1; i ++;)
           {
            n = n *( n-i );
           }
      }
  return n;
}
